CREACION DE LAS TABLAS:

CREATE TABLE PROFESOR (DNI_PROFESOR VARCHAR(9) PRIMARY KEY NOT NULL, NOMBRE_PROFESOR VARCHAR(30) NOT NULL, APELLIDO_PROFESOR VARCHAR(50) NOT NULL, SUELDO INT NOT NULL);
CREATE TABLE ASIGNATURA (ASIGNATURA INT PRIMARY KEY NOT NULL, NOMBRE_ASIGNATURA VARCHAR(50) NOT NULL, DESCRIPCION VARCHAR(200) NOT NULL, DNI_PROFESOR_ASIGNATURA VARCHAR(9) NOT NULL, FOREIGN KEY (DNI_PROFESOR_ASIGNATURA) REFERENCES PROFESOR (DNI_PROFESOR));
CREATE TABLE ALUMNO (DNI_ALUMNO VARCHAR(9) PRIMARY KEY NOT NULL, NOMBRE_ALUMNO VARCHAR(30) NOT NULL, APELLIDO_ALUMNO VARCHAR(50) NOT NULL, ASIGNATURA_IMPARTIDA INT NOT NULL, FOREIGN KEY (ASIGNATURA_IMPARTIDA) REFERENCES ASIGNATURA (ASIGNATURA));
